42  struct ContainsMatcher : MatcherBase<std::vector<T, AllocMatch>> {
43 
44  ContainsMatcher(std::vector<T, AllocComp> const &comparator) : m_comparator( comparator ) {}
45 
46  bool match(std::vector<T, AllocMatch> const &v) const override {
47  // !TBD: see note in EqualsMatcher
48  if (m_comparator.size() > v.size())
49  return false;
50  for (auto const& comparator : m_comparator) {
51  auto present = false;
52  for (const auto& el : v) {
53  if (el == comparator) {
54  present = true;
55  break;
56  }
57  }

61  }

63  }
