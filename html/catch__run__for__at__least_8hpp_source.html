48  template <typename Clock, typename Fun>
49  TimingOf<Clock, Fun, run_for_at_least_argument_t<Clock, Fun>> run_for_at_least(ClockDuration<Clock> how_long, int seed, Fun&& fun) {
50  auto iters = seed;
51  while (iters < (1 << 30)) {
52  auto&& Timing = measure_one<Clock>(fun, iters, is_callable<Fun(Chronometer)>());
53 
54  if (Timing.elapsed >= how_long) {
55  return { Timing.elapsed, std::move(Timing.result), iters };
56  }
57  iters *= 2;
58  }
59  Catch::throw_exception(optimized_away_error{});
