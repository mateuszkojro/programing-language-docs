177  void sectionStarting( SectionInfo const& sectionInfo ) override {
178  SectionStats incompleteStats( sectionInfo, Counts(), 0, false );
179  std::shared_ptr<SectionNode> node;
180  if( m_sectionStack.empty() ) {
181  if( !m_rootSection )
182  m_rootSection = std::make_shared<SectionNode>( incompleteStats );
183  node = m_rootSection;
184  }
185  else {
186  SectionNode& parentNode = *m_sectionStack.back();
187  auto it =
188  std::find_if( parentNode.childSections.begin(),
189  parentNode.childSections.end(),
190  BySectionInfo( sectionInfo ) );
191  if( it == parentNode.childSections.end() ) {
192  node = std::make_shared<SectionNode>( incompleteStats );
193  parentNode.childSections.push_back( node );
194  }
195  else
196  node = *it;
197  }
198  m_sectionStack.push_back( node );
199  m_deepestSection = std::move(node);
200  }

204  bool assertionEnded(AssertionStats const& assertionStats) override {
205  assert(!m_sectionStack.empty());
206  // AssertionResult holds a pointer to a temporary DecomposedExpression,
207  // which getExpandedExpression() calls to build the expression string.
208  // Our section stack copy of the assertionResult will likely outlive the
209  // temporary, so it must be expanded or discarded now to avoid calling
210  // a destroyed object later.
211  prepareExpandedExpression(const_cast<AssertionResult&>( assertionStats.assertionResult ) );
212  SectionNode& sectionNode = *m_sectionStack.back();
213  sectionNode.assertions.push_back(assertionStats);
214  return true;
215  }
216  void sectionEnded(SectionStats const& sectionStats) override {
217  assert(!m_sectionStack.empty());
218  SectionNode& node = *m_sectionStack.back();
219  node.stats = sectionStats;
220  m_sectionStack.pop_back();
221  }
222  void testCaseEnded(TestCaseStats const& testCaseStats) override {
223  auto node = std::make_shared<TestCaseNode>(testCaseStats);
224  assert(m_sectionStack.size() == 0);
225  node->children.push_back(m_rootSection);
226  m_testCases.push_back(node);
227  m_rootSection.reset();
228 
229  assert(m_deepestSection);
230  m_deepestSection->stdOut = testCaseStats.stdOut;
231  m_deepestSection->stdErr = testCaseStats.stdErr;
232  }
233  void testGroupEnded(TestGroupStats const& testGroupStats) override {
234  auto node = std::make_shared<TestGroupNode>(testGroupStats);
235  node->children.swap(m_testCases);
236  m_testGroups.push_back(node);
237  }
238  void testRunEnded(TestRunStats const& testRunStats) override {
239  auto node = std::make_shared<TestRunNode>(testRunStats);
240  node->children.swap(m_testGroups);
241  m_testRuns.push_back(node);
242  testRunEndedCumulative();
243  }
