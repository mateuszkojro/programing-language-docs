26  std::string getCurrentTimestamp() {
27  // Beware, this is not reentrant because of backward compatibility issues
28  // Also, UTC only, again because of backward compatibility (%z is C++11)
29  time_t rawtime;
30  std::time(&rawtime);
31  auto const timeStampSize = sizeof("2017-01-16T17:06:45Z");
32 
33 #ifdef _MSC_VER
34  std::tm timeInfo = {};
35  gmtime_s(&timeInfo, &rawtime);
36 #else
37  std::tm* timeInfo;
38  timeInfo = std::gmtime(&rawtime);
39 #endif
40 
41  char timeStamp[timeStampSize];
42  const char * const fmt = "%Y-%m-%dT%H:%M:%SZ";
43 
44 #ifdef _MSC_VER
45  std::strftime(timeStamp, timeStampSize, fmt, &timeInfo);
46 #else
47  std::strftime(timeStamp, timeStampSize, fmt, timeInfo);
48 #endif
49  return std::string(timeStamp, timeStampSize-1);
50  }

128  void JunitReporter::writeGroup( TestGroupNode const& groupNode, double suiteTime ) {
129  XmlWriter::ScopedElement e = xml.scopedElement( "testsuite" );
130 
131  TestGroupStats const& stats = groupNode.value;
132  xml.writeAttribute( "name", stats.groupInfo.name );
133  xml.writeAttribute( "errors", unexpectedExceptions );
134  xml.writeAttribute( "failures", stats.totals.assertions.failed-unexpectedExceptions );
135  xml.writeAttribute( "tests", stats.totals.assertions.total() );
136  xml.writeAttribute( "hostname", "tbd" ); // !TBD
137  if( m_config->showDurations() == ShowDurations::Never )
138  xml.writeAttribute( "time", "" );
139  else
140  xml.writeAttribute( "time", formatDuration( suiteTime ) );
141  xml.writeAttribute( "timestamp", getCurrentTimestamp() );
142 
143  // Write properties if there are any
144  if (m_config->hasTestFilters() || m_config->rngSeed() != 0) {
145  auto properties = xml.scopedElement("properties");
146  if (m_config->hasTestFilters()) {
147  xml.scopedElement("property")
148  .writeAttribute("name", "filters")
149  .writeAttribute("value", serializeFilters(m_config->getTestsOrTags()));
150  }
151  if (m_config->rngSeed() != 0) {
152  xml.scopedElement("property")
153  .writeAttribute("name", "random-seed")
154  .writeAttribute("value", m_config->rngSeed());
155  }
156  }
157 
158  // Write test cases
159  for( auto const& child : groupNode.children )
160  writeTestCase( *child );
161 
162  xml.scopedElement( "system-out" ).writeText( trim( stdOutForSuite ), XmlFormatting::Newline );
163  xml.scopedElement( "system-err" ).writeText( trim( stdErrForSuite ), XmlFormatting::Newline );
164  }

166  void JunitReporter::writeTestCase( TestCaseNode const& testCaseNode ) {
167  TestCaseStats const& stats = testCaseNode.value;
168 
169  // All test cases have exactly one section - which represents the
170  // test case itself. That section may have 0-n nested sections
171  assert( testCaseNode.children.size() == 1 );
172  SectionNode const& rootSection = *testCaseNode.children.front();
173 
174  std::string className = stats.testInfo.className;
175 
176  if( className.empty() ) {
177  className = fileNameTag(stats.testInfo.tags);
178  if ( className.empty() )
179  className = "global";
180  }
181 
182  if ( !m_config->name().empty() )
183  className = m_config->name() + "." + className;
184 
185  writeSection( className, "", rootSection, stats.testInfo.okToFail() );
186  }

188  void JunitReporter::writeSection( std::string const& className,
189  std::string const& rootName,
190  SectionNode const& sectionNode,
191  bool testOkToFail) {
192  std::string name = trim( sectionNode.stats.sectionInfo.name );
193  if( !rootName.empty() )
194  name = rootName + '/' + name;
195 
196  if( !sectionNode.assertions.empty() ||
197  !sectionNode.stdOut.empty() ||
198  !sectionNode.stdErr.empty() ) {
199  XmlWriter::ScopedElement e = xml.scopedElement( "testcase" );
200  if( className.empty() ) {
201  xml.writeAttribute( "classname", name );
202  xml.writeAttribute( "name", "root" );
203  }
204  else {
205  xml.writeAttribute( "classname", className );
206  xml.writeAttribute( "name", name );
207  }
208  xml.writeAttribute( "time", formatDuration( sectionNode.stats.durationInSeconds ) );
209  // This is not ideal, but it should be enough to mimic gtest's
210  // junit output.
211  // Ideally the JUnit reporter would also handle `skipTest`
212  // events and write those out appropriately.
213  xml.writeAttribute( "status", "run" );
214 
215  if (sectionNode.stats.assertions.failedButOk) {
216  xml.scopedElement("skipped")
217  .writeAttribute("message", "TEST_CASE tagged with !mayfail");
218  }
219 
220  writeAssertions( sectionNode );
221 
222 
223  if( !sectionNode.stdOut.empty() )
224  xml.scopedElement( "system-out" ).writeText( trim( sectionNode.stdOut ), XmlFormatting::Newline );
225  if( !sectionNode.stdErr.empty() )
226  xml.scopedElement( "system-err" ).writeText( trim( sectionNode.stdErr ), XmlFormatting::Newline );
227  }
228  for( auto const& childNode : sectionNode.childSections )
229  if( className.empty() )
230  writeSection( name, "", *childNode, testOkToFail );
231  else
232  writeSection( className, name, *childNode, testOkToFail );
233  }

240  void JunitReporter::writeAssertion( AssertionStats const& stats ) {
241  AssertionResult const& result = stats.assertionResult;
242  if( !result.isOk() ) {
243  std::string elementName;
244  switch( result.getResultType() ) {
245  case ResultWas::ThrewException:
246  case ResultWas::FatalErrorCondition:
247  elementName = "error";
248  break;
249  case ResultWas::ExplicitFailure:
250  case ResultWas::ExpressionFailed:
251  case ResultWas::DidntThrowException:
252  elementName = "failure";
253  break;
254 
255  // We should never see these here:
256  case ResultWas::Info:
257  case ResultWas::Warning:
258  case ResultWas::Ok:
259  case ResultWas::Unknown:
260  case ResultWas::FailureBit:
261  case ResultWas::Exception:
262  elementName = "internalError";
263  break;
264  }
265 
266  XmlWriter::ScopedElement e = xml.scopedElement( elementName );
267 
268  xml.writeAttribute( "message", result.getExpression() );
269  xml.writeAttribute( "type", result.getTestMacroName() );
270 
271  ReusableStringStream rss;
272  if (stats.totals.assertions.total() > 0) {
273  rss << "FAILED" << ":\n";
274  if (result.hasExpression()) {
275  rss << " ";
276  rss << result.getExpressionInMacro();
277  rss << '\n';
278  }
279  if (result.hasExpandedExpression()) {
280  rss << "with expansion:\n";
281  rss << Column(result.getExpandedExpression()).indent(2) << '\n';
282  }
283  } else {
284  rss << '\n';
285  }
286 
287  if( !result.getMessage().empty() )
288  rss << result.getMessage() << '\n';
289  for( auto const& msg : stats.infoMessages )
290  if( msg.type == ResultWas::Info )
291  rss << msg.message << '\n';
292 
293  rss << "at " << result.getSourceInfo();
294  xml.writeText( rss.str(), XmlFormatting::Newline );
295  }
296  }
