24  struct TestHasher {
25  using hash_t = uint64_t;
26 
27  explicit TestHasher( hash_t hashSuffix ):
28  m_hashSuffix{ hashSuffix } {}
29 
30  uint32_t operator()( TestCase const& t ) const {
31  // FNV-1a hash with multiplication fold.
32  const hash_t prime = 1099511628211u;
33  hash_t hash = 14695981039346656037u;
34  for ( const char c : t.name ) {
35  hash ^= c;
36  hash *= prime;
37  }
38  hash ^= m_hashSuffix;
39  hash *= prime;
40  const uint32_t low{ static_cast<uint32_t>( hash ) };
41  const uint32_t high{ static_cast<uint32_t>( hash >> 32 ) };
42  return low * high;
43  }
44 
45  private:
46  hash_t m_hashSuffix;
47  };

51  std::vector<TestCase> sortTests( IConfig const& config, std::vector<TestCase> const& unsortedTestCases ) {
52  switch( config.runOrder() ) {
53  case RunTests::InDeclarationOrder:
54  // already in declaration order
55  break;
56 
57  case RunTests::InLexicographicalOrder: {
58  std::vector<TestCase> sorted = unsortedTestCases;
59  std::sort( sorted.begin(), sorted.end() );
60  return sorted;
61  }
62 
63  case RunTests::InRandomOrder: {
64  seedRng( config );
65  TestHasher h{ config.rngSeed() };
66 
67  using hashedTest = std::pair<TestHasher::hash_t, TestCase const*>;
68  std::vector<hashedTest> indexed_tests;
69  indexed_tests.reserve( unsortedTestCases.size() );
70 
71  for (auto const& testCase : unsortedTestCases) {
72  indexed_tests.emplace_back(h(testCase), &testCase);
73  }
74 
75  std::sort(indexed_tests.begin(), indexed_tests.end(),
76  [](hashedTest const& lhs, hashedTest const& rhs) {
77  if (lhs.first == rhs.first) {
78  return lhs.second->name < rhs.second->name;
79  }
80  return lhs.first < rhs.first;
81  });
82 
83  std::vector<TestCase> sorted;
84  sorted.reserve( indexed_tests.size() );
85 
86  for (auto const& hashed : indexed_tests) {
87  sorted.emplace_back(*hashed.second);
88  }
89 
90  return sorted;
91  }
92  }
93  return unsortedTestCases;
94  }
