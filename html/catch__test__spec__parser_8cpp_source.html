37  bool TestSpecParser::visitChar( char c ) {
38  if( (m_mode != EscapedName) && (c == '\\') ) {
39  escape();
40  addCharToPattern(c);
41  return true;
42  }else if((m_mode != EscapedName) && (c == ',') ) {
43  return separate();
44  }
45 
46  switch( m_mode ) {
47  case None:
48  if( processNoneChar( c ) )
49  return true;
50  break;
51  case Name:
52  processNameChar( c );
53  break;
54  case EscapedName:
55  endMode();
56  addCharToPattern(c);
57  return true;
58  default:
59  case Tag:
60  case QuotedName:

64  }

67  if( !isControlChar( c ) ) {
